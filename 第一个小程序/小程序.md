        wxml  -->  html
        wxss  -->  css
        json  当前页面的配置文件
        js    只对当前所在的页面生效

    --list  最少五个,最多两个

    "text":     tabBar导航对应文字
    "iconPath": 对应的图标
    "selectedIconPath": 选中之后的图标
